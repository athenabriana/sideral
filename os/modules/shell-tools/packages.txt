# helix — modal terminal editor (treesitter, LSP, multi-cursor by default).
# Set as $EDITOR by sideral-cli-init.{sh,fish} so git, sudoedit, mise
# edit, etc. drop into it by default.
helix
# fish — friendly interactive shell. First-class syntax highlighting,
# autosuggestions from history, and smarter tab completion than bash
# all built in (no config). Sideral ships parallel bash/fish/zsh init
# wiring (sideral-cli-init.{sh,fish,zsh}); switch via `ujust chsh fish`.
fish
# zsh — third interactive-shell option. Has its own ecosystem
# (oh-my-zsh, prezto, etc.) but works fine vanilla too. Sideral's
# /etc/zshrc + /etc/zsh/sideral-cli-init.zsh wire it up the same as
# the other two shells. Switch via `ujust chsh zsh`.
zsh
