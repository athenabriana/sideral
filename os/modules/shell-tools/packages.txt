# helix — modal terminal editor (treesitter, LSP, multi-cursor by default).
# Set as $EDITOR by sideral-cli-init.{sh,fish} so git, sudoedit, mise
# edit, etc. drop into it by default.
helix
# fish — friendly interactive shell. First-class syntax highlighting,
# autosuggestions from history, and smarter tab completion than bash
# all built in (no config). Sideral ships parallel bash/fish/zsh init
# wiring (sideral-cli-init.{sh,fish,zsh}); switch via `ujust chsh fish`.
fish
# zsh — third interactive-shell option. Switch via `ujust chsh zsh`.
# Sideral's /etc/zshrc + /etc/zsh/sideral-cli-init.zsh wire it up
# the same as bash and fish.
zsh
# zsh-syntax-highlighting + zsh-autosuggestions — bring zsh to
# parity with fish's two killer interactive features (red-on-invalid
# command coloring + greyed-out autosuggestions from history). Both
# from Fedora main, source-loaded by sideral-cli-init.zsh — no plugin
# manager needed for two source lines. Users who want bigger plugin
# stacks (oh-my-zsh, prezto, zinit) install at user level on top.
zsh-syntax-highlighting
zsh-autosuggestions
